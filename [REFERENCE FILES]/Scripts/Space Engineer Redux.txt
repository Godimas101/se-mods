// In custom data of cockpit or remote use: SER
// In custom data of LCDs use: SERLCD
// Alerts are optional and configurable.
// SERSound, SERLight, SERTimer customdata are used for the alert actions.

public Program()
{
    Runtime.UpdateFrequency = UpdateFrequency.Update10;
}

// ------------------- ALERT CONFIG -------------------

// Enable alert for Critical UP TWR?
const bool ALERT_UP_ENABLED = true;
// Threshold for Critical UP TWR (default = 1.35)
const double CRITICAL_TWR_UP = 1.35;

//------------------------------------------------------------

// Enable alert for Critical BACKWARDS TWR?
const bool ALERT_BACK_ENABLED = false;
// Threshold for Critical BACKWARDS TWR (default = 1.35)
const double CRITICAL_TWR_BACK = 1.35;

// ---------------------- ALERT END ----------------------

const string SERSOUND_CDATA = "SERSound";
const string SERLIGHT_CDATA = "SERLight";
const string SERTIMER_CDATA = "SERTimer";

const string cockpitMasterTHRKeyword = "SER";
const string lcdMasterKeyword = "SERLCD";
const double G_DEFAULT = 11.77; // 1.2g Pertam
const int LINE_WIDTH = 30;

const int LCD_COL_CHARS = 35;
const string COL_SEPARATOR = " | ";

// Sound alert interval 10 ticks = 1 sec
const int SOUND_RESTART_TICKS = 15; 

IMyShipController masterController;
IMyTerminalBlock masterBlock; // terminal block reference for grid/worldmatrix
IMyCockpit masterCockpit; // non-null only when master is a cockpit

bool alertActive = false;
int soundRestartCounter = 0;

class PlanetData
{
    public string Name;
    public double Gravity;
    public PlanetData(string name, double gravity) { Name = name; Gravity = gravity; }
}

PlanetData[] planets = new PlanetData[]
{
    new PlanetData("Earth", 9.81),
    new PlanetData("Mars", 8.83),
    new PlanetData("Alien", 10.79),
    new PlanetData("Moon", 2.45),
    new PlanetData("Pertam", 11.77),
    new PlanetData("Triton", 10.79)
};

// formatting helpers
string FormatLine(string label, string value)
{
    string line = value == "" ? label : label + ": " + value;
    if (line.Length > LINE_WIDTH)
        line = line.Substring(0, LINE_WIDTH);
    return line;
}

string CenterTextVertical(IMyTextSurface lcd, string text)
{
    string[] lines = text.Split('\n');
    int totalLines = lines.Length;

    float lcdHeightPx = lcd.SurfaceSize.Y;
    float lineHeightPx = lcd.FontSize * 30;
    int maxLines = (int)(lcdHeightPx / lineHeightPx);

    if (totalLines >= maxLines) return text;

    int paddingLines = (maxLines - totalLines) / 2;

    string padding = "";
    for (int i = 0; i < paddingLines; i++) padding += "\n";

    return padding + text;
}

string TruncateOrPad(string s, int width)
{
    if (s.Length > width) return s.Substring(0, width);
    return s.PadRight(width);
}

string CenterTextToWidth(string text, int width)
{
    if (text.Length >= width) return text.Substring(0, width);
    int leftPad = (width - text.Length) / 2;
    if (leftPad < 0) leftPad = 0;
    return new string(' ', leftPad) + text;
}
// ---------------------------------------------------

public void Main(string argument, UpdateType updateSource)
{
    // find master controller (cockpit / remote / control seat) on same grid
    var controllers = new List<IMyShipController>();
    GridTerminalSystem.GetBlocksOfType<IMyShipController>(controllers, c => c.CubeGrid == Me.CubeGrid);

    masterController = null;
    masterBlock = null;
    masterCockpit = null;

    foreach (var c in controllers)
    {
        var term = c as IMyTerminalBlock;
        if (term != null && term.CustomData != null && term.CustomData.ToUpper().Contains(cockpitMasterTHRKeyword.ToUpper()))
        {
            masterController = c;
            masterBlock = term;
            masterCockpit = term as IMyCockpit; // will be null if remote control
            break;
        }
    }

    if (masterController == null)
    {
        Echo("No master controller found (set CustomData to '" + cockpitMasterTHRKeyword + "')");
        return;
    }

    // configure cockpit surfaces ONLY if we actually have a cockpit
    if (masterCockpit != null)
    {
        for (int i = 0; i < masterCockpit.SurfaceCount; i++)
        {
            var surf = masterCockpit.GetSurface(i);
            surf.ContentType = VRage.Game.GUI.TextPanel.ContentType.TEXT_AND_IMAGE;
            surf.Font = "Debug";
            surf.Alignment = VRage.Game.GUI.TextPanel.TextAlignment.LEFT;
            surf.FontColor = new VRageMath.Color(255, 191, 0);
            surf.BackgroundColor = new VRageMath.Color(10, 10, 10);
        }
    }

    // mass & gravity (use ship controller API)
    var shipMass = masterController.CalculateShipMass();
    double shipMassKg = shipMass.PhysicalMass;
    double gravity = (double)masterController.GetNaturalGravity().Length();
    if (gravity <= 0) gravity = G_DEFAULT;
    double weight = shipMassKg * gravity;           // Newtons
    double weightT = shipMassKg / 1000.0;           // tonnes (metric)

    // get thrusters on same grid
    var thrusters = new List<IMyThrust>();
    GridTerminalSystem.GetBlocksOfType(thrusters, t => t.CubeGrid == masterBlock.CubeGrid);

    double upMax = 0.0, upCurrent = 0.0;
    double backMax = 0.0, backCurrent = 0.0;

    // Counters for working / not working
    int upWorking = 0, upNotWorking = 0;
    int backWorking = 0, backNotWorking = 0;

    foreach (var t in thrusters)
    {
        // determine local direction relative to master controller
        VRageMath.Vector3D thrustDir = t.WorldMatrix.Forward;
        VRageMath.Vector3D localDir = VRageMath.Vector3D.TransformNormal(thrustDir, VRageMath.MatrixD.Transpose(masterBlock.WorldMatrix));
        localDir.Normalize();

        // check direction: up (Y < -0.5) or back (Z < -0.5)
        if (localDir.Y < -0.5) // UP thrusters
        {
            bool active = ThrusterIsActive(t);
            if (active)
            {
                upMax += t.MaxEffectiveThrust;
                upCurrent += t.CurrentThrust;
                upWorking++;
            }
            else
            {
                upNotWorking++;
            }
        }
        else if (localDir.Z < -0.5) // BACK thrusters
        {
            bool active = ThrusterIsActive(t);
            if (active)
            {
                backMax += t.MaxEffectiveThrust;
                backCurrent += t.CurrentThrust;
                backWorking++;
            }
            else
            {
                backNotWorking++;
            }
        }
    }

    double upMaxT = gravity > 0 ? upMax / gravity / 1000.0 : 0.0;
    double backMaxT = gravity > 0 ? backMax / gravity / 1000.0 : 0.0;

    double upUsage = upMax > 0.0 ? (upCurrent / upMax) * 100.0 : 0.0;
    double backUsage = backMax > 0.0 ? (backCurrent / backMax) * 100.0 : 0.0;

    double upTWR = weight > 0.0 ? upMax / weight : 0.0;
    double backTWR = weight > 0.0 ? backMax / weight : 0.0;

    double upMargin = (upTWR - 1.0) * weightT;
    double backMargin = (backTWR - 1.0) * weightT;

    // ship speed
    double velocity = masterController.GetShipVelocities().LinearVelocity.Length();

    // acceleration up
    double upAccel = shipMassKg > 0.0 ? upMax / shipMassKg : 0.0;
    double upBrakeDist = upAccel > 0.0 ? ((velocity * velocity) / (2.0 * upAccel)) / 1000.0 : -1.0;
    double upBrakeTime = upAccel > 0.0 ? velocity / upAccel : -1.0;

    // acceleration back
    double backAccel = shipMassKg > 0.0 ? backMax / shipMassKg : 0.0;
    double backBrakeDist = backAccel > 0.0 ? ((velocity * velocity) / (2.0 * backAccel)) / 1000.0 : -1.0;
    double backBrakeTime = backAccel > 0.0 ? velocity / backAccel : -1.0;

    // determine alert conditions now so we can display them in the cockpit shipinfo
    bool wantUpAlert = ALERT_UP_ENABLED && upTWR < CRITICAL_TWR_UP;
    bool wantBackAlert = ALERT_BACK_ENABLED && backTWR < CRITICAL_TWR_BACK;

    string alertCriticalSummary;
    if (wantUpAlert || wantBackAlert)
    {
        alertCriticalSummary = (wantUpAlert ? "UP" : "") + (wantBackAlert ? (wantUpAlert ? " BACK" : "BACK") : "");
    }
    else alertCriticalSummary = "none";

    // display texts
    string upText = FormatLine("UP THRUSTERS", "") + "\n" +
                    FormatLine("Max Lift", upMaxT.ToString("0.00") + " t") + "\n" +
                    FormatLine("Usage", upUsage.ToString("0") + "%") + "\n" +
                    FormatLine("TWR", upTWR.ToString("0.00")) + "\n" +
                    FormatLine("Margin", upMargin.ToString("0.00") + " t") + "\n" +
                    FormatLine("Accel", upAccel.ToString("0.00") + " m/s²") + "\n" +
                    FormatLine("Brake Dist", upBrakeDist >= 0 ? upBrakeDist.ToString("0.00") + " km" : "N/A") + "\n" +
                    FormatLine("Brake Time", upBrakeTime >= 0 ? upBrakeTime.ToString("0.0") + " s" : "N/A") + "\n" +
                    FormatLine("Working", upWorking + " / " + (upWorking + upNotWorking));

    string massText = FormatLine("SHIP INFO", "") + "\n" +
                      FormatLine("Weight", weightT.ToString("0.00") + " t") + "\n" +
                      FormatLine("Gravity", gravity.ToString("0.00") + " m/s²") + "\n" +
                      FormatLine("Speed", velocity.ToString("0.0") + " m/s") + "\n" +
                      FormatLine("Alert TWR", alertCriticalSummary);

    string backText = FormatLine("BACK THRUSTERS", "") + "\n" +
                      FormatLine("Max Push", backMaxT.ToString("0.00") + " t") + "\n" +
                      FormatLine("Usage", backUsage.ToString("0") + "%") + "\n" +
                      FormatLine("TWR", backTWR.ToString("0.00")) + "\n" +
                      FormatLine("Margin", backMargin.ToString("0.00") + " t") + "\n" +
                      FormatLine("Accel", backAccel.ToString("0.00") + " m/s²") + "\n" +
                      FormatLine("Brake Dist", backBrakeDist >= 0 ? backBrakeDist.ToString("0.00") + " km" : "N/A") + "\n" +
                      FormatLine("Brake Time", backBrakeTime >= 0 ? backBrakeTime.ToString("0.0") + " s" : "N/A") + "\n" +
                      FormatLine("Working", backWorking + " / " + (backWorking + backNotWorking));

    bool isSmallGrid = masterBlock.CubeGrid.GridSizeEnum == VRage.Game.MyCubeSize.Small;

    if (masterCockpit != null)
    {
        if (isSmallGrid)
        {
            if (masterCockpit.SurfaceCount > 0) masterCockpit.GetSurface(0).WriteText(CenterTextVertical(masterCockpit.GetSurface(0), upText));
            if (masterCockpit.SurfaceCount > 1) masterCockpit.GetSurface(1).WriteText(CenterTextVertical(masterCockpit.GetSurface(1), massText));
            if (masterCockpit.SurfaceCount > 2) masterCockpit.GetSurface(2).WriteText(CenterTextVertical(masterCockpit.GetSurface(2), backText));
        }
        else
        {
            if (masterCockpit.SurfaceCount > 0) masterCockpit.GetSurface(0).WriteText(CenterTextVertical(masterCockpit.GetSurface(0), massText));
            if (masterCockpit.SurfaceCount > 1) masterCockpit.GetSurface(1).WriteText(CenterTextVertical(masterCockpit.GetSurface(1), upText));
            if (masterCockpit.SurfaceCount > 2) masterCockpit.GetSurface(2).WriteText(CenterTextVertical(masterCockpit.GetSurface(2), backText));
        }

        // --- Planet Lift Status cockpit (only for cockpits with extra surface)
        if (masterCockpit.SurfaceCount > 3)
        {
            var sbPlanet = new System.Text.StringBuilder();
            const int PLANET_LINE_WIDTH = 35;
            sbPlanet.AppendLine(CenterTextToWidth("PLANET LIFT STATUS", PLANET_LINE_WIDTH));
            sbPlanet.AppendLine("");
            foreach (var p in planets)
            {
                double twr = upMax / (shipMassKg * p.Gravity);
                string status = twr > 1.2 ? "Ok" : (twr >= 1.05 ? "Not Recommended" : "Not Ok");
                sbPlanet.AppendLine(p.Name + ": " + status + " (" + twr.ToString("0.00") + ")");
            }
            masterCockpit.GetSurface(3).WriteText(CenterTextVertical(masterCockpit.GetSurface(3), sbPlanet.ToString()));
        }
    }

    // --- External LCDs ---
    var lcds = new List<IMyTextPanel>();
    GridTerminalSystem.GetBlocksOfType(lcds, l => l.CustomData != null && l.CustomData.ToUpper().Contains(lcdMasterKeyword.ToUpper()));

    foreach (var lcd in lcds)
    {
        lcd.ContentType = VRage.Game.GUI.TextPanel.ContentType.TEXT_AND_IMAGE;
        lcd.Font = "Monospace";
        lcd.Alignment = VRage.Game.GUI.TextPanel.TextAlignment.LEFT;
        lcd.FontColor = new VRageMath.Color(255, 191, 0);
        lcd.BackgroundColor = new VRageMath.Color(10, 10, 10);

        var sb = new System.Text.StringBuilder();

        int fullWidth = LCD_COL_CHARS * 2 + COL_SEPARATOR.Length;
        sb.AppendLine(CenterTextToWidth("SHIP INFO", fullWidth));
        sb.AppendLine(CenterTextToWidth("Weight: " + weightT.ToString("0.00") + " t", fullWidth));
        sb.AppendLine(CenterTextToWidth("Gravity: " + gravity.ToString("0.00") + " m/s²", fullWidth));
        sb.AppendLine(CenterTextToWidth("Speed: " + velocity.ToString("0.0") + " m/s", fullWidth));

        sb.AppendLine(new string('_', fullWidth));

        string leftHeader = TruncateOrPad("UP Thrusters", LCD_COL_CHARS);
        string rightHeader = TruncateOrPad("BACK Thrusters", LCD_COL_CHARS);
        sb.AppendLine(leftHeader + COL_SEPARATOR + rightHeader);

        string[] leftLines = {
            "Max Lift: " + upMaxT.ToString("0.00") + " t",
            "Usage: " + upUsage.ToString("0") + "%",
            "TWR: " + upTWR.ToString("0.00"),
            "Margin: " + upMargin.ToString("0.00") + " t",
            "Accel: " + upAccel.ToString("0.00") + " m/s²",
            "Brake Dist: " + (upBrakeDist >= 0 ? upBrakeDist.ToString("0.00") + " km" : "N/A"),
            "Brake Time: " + (upBrakeTime >= 0 ? upBrakeTime.ToString("0.0") + " s" : "N/A"),
            "Working: " + upWorking + "/" + (upWorking + upNotWorking)
        };

        string[] rightLines = {
            "Max Push: " + backMaxT.ToString("0.00") + " t",
            "Usage: " + backUsage.ToString("0") + "%",
            "TWR: " + backTWR.ToString("0.00"),
            "Margin: " + backMargin.ToString("0.00") + " t",
            "Accel: " + backAccel.ToString("0.00") + " m/s²",
            "Brake Dist: " + (backBrakeDist >= 0 ? backBrakeDist.ToString("0.00") + " km" : "N/A"),
            "Brake Time: " + (backBrakeTime >= 0 ? backBrakeTime.ToString("0.0") + " s" : "N/A"),
            "Working: " + backWorking + "/" + (backWorking + backNotWorking)
        };

        int maxLines = Math.Max(leftLines.Length, rightLines.Length);
        for (int i = 0; i < maxLines; i++)
        {
            string l = i < leftLines.Length ? leftLines[i] : "";
            string r = i < rightLines.Length ? rightLines[i] : "";
            sb.AppendLine(TruncateOrPad(l, LCD_COL_CHARS) + COL_SEPARATOR + TruncateOrPad(r, LCD_COL_CHARS));
        }

        sb.AppendLine(new string('_', fullWidth));
        sb.AppendLine("");
        sb.AppendLine(CenterTextToWidth("PLANET LIFT STATUS", fullWidth));

        for (int i = 0; i < planets.Length; i += 2)
        {
            string left = FormatPlanetResult(planets[i], upMax, shipMassKg);
            string right = i + 1 < planets.Length ? FormatPlanetResult(planets[i + 1], upMax, shipMassKg) : "";
            sb.AppendLine(TruncateOrPad(left, LCD_COL_CHARS) + COL_SEPARATOR + TruncateOrPad(right, LCD_COL_CHARS));
        }

        // Add alert summary line to external LCD
        string alertSummary = "Alert Critical TWR: " + alertCriticalSummary;
        sb.AppendLine("");
        sb.AppendLine(CenterTextToWidth(alertSummary, fullWidth));

        lcd.WriteText(CenterTextVertical(lcd, sb.ToString()));
    }

    // ------------------ ALERT HANDLING ------------------
    bool wantAlert = wantUpAlert || wantBackAlert;

    if (wantAlert && !alertActive)
    {
        // entering alert state
        ActivateAlert(wantUpAlert, wantBackAlert);
        alertActive = true;
        soundRestartCounter = 0;
    }
    else if (!wantAlert && alertActive)
    {
        // exiting alert state
        DeactivateAlert();
        alertActive = false;
    }

    if (alertActive)
    {
        soundRestartCounter++;
        if (soundRestartCounter >= SOUND_RESTART_TICKS)
        {
            soundRestartCounter = 0;
            ForcePlayAllSounds();
        }
    }

    // PB console output
    Echo("Up Thrusters: Working=" + upWorking + ", NotWorking=" + upNotWorking);
    Echo("Back Thrusters: Working=" + backWorking + ", NotWorking=" + backNotWorking);
    Echo("TWR (up): " + upTWR.ToString("0.00") + " (limit " + CRITICAL_TWR_UP.ToString("0.00") + (ALERT_UP_ENABLED ? ", enabled" : ", disabled") + ")");
    Echo("TWR (back): " + backTWR.ToString("0.00") + " (limit " + CRITICAL_TWR_BACK.ToString("0.00") + (ALERT_BACK_ENABLED ? ", enabled" : ", disabled") + ")");
    Echo("Alert: " + (alertActive ? "ACTIVE" : "OFF"));
}

// ----------------- Helpers for alert devices -----------------

void ActivateAlert(bool upTriggered, bool backTriggered)
{
    Echo("Activating ALERT: " + (upTriggered ? "UP " : "") + (backTriggered ? "BACK" : ""));

    // sounds
    ForcePlayAllSounds();

    // lights
    var lights = new List<IMyLightingBlock>();
    GridTerminalSystem.GetBlocksOfType(lights, l =>
        l != null &&
        l.CubeGrid == masterBlock.CubeGrid &&
        l.CustomData != null &&
        l.CustomData.ToUpper().Contains(SERLIGHT_CDATA.ToUpper())
    );

    foreach (var light in lights)
    {
        try { light.Enabled = true; }
        catch { }
    }

    // timers
    var timers = new List<IMyTimerBlock>();
    GridTerminalSystem.GetBlocksOfType(timers, t =>
        t != null &&
        t.CubeGrid == masterBlock.CubeGrid &&
        t.CustomData != null &&
        t.CustomData.ToUpper().Contains(SERTIMER_CDATA.ToUpper())
    );

    foreach (var timer in timers)
    {
        try { timer.Trigger(); }
        catch
        {
            try { ((IMyTerminalBlock)timer).ApplyAction("TriggerNow"); } catch { }
        }
    }
}

void DeactivateAlert()
{
    Echo("Deactivating ALERT (recovered)");

    StopAllSounds();

    // lights
    var lights = new List<IMyLightingBlock>();
    GridTerminalSystem.GetBlocksOfType(lights, l =>
        l != null &&
        l.CubeGrid == masterBlock.CubeGrid &&
        l.CustomData != null &&
        l.CustomData.ToUpper().Contains(SERLIGHT_CDATA.ToUpper())
    );

    foreach (var light in lights)
    {
        try { light.Enabled = false; }
        catch { }
    }

    // timers
    var timers = new List<IMyTimerBlock>();
    GridTerminalSystem.GetBlocksOfType(timers, t =>
        t != null &&
        t.CubeGrid == masterBlock.CubeGrid &&
        t.CustomData != null &&
        t.CustomData.ToUpper().Contains(SERTIMER_CDATA.ToUpper())
    );

    foreach (var timer in timers)
    {
        try { timer.Trigger(); }
        catch
        {
            try { ((IMyTerminalBlock)timer).ApplyAction("TriggerNow"); } catch { }
        }
    }
}

void ForcePlayAllSounds()
{
    var soundBlocks = new List<IMySoundBlock>();
    GridTerminalSystem.GetBlocksOfType(soundBlocks, b =>
        b != null &&
        b.CubeGrid == masterBlock.CubeGrid &&
        b.CustomData != null &&
        b.CustomData.ToUpper().Contains(SERSOUND_CDATA.ToUpper())
    );

    foreach (var sb in soundBlocks)
    {
        try
        {
            // volume to 100%
            try { sb.Volume = 1f; } catch { try { sb.SetValueFloat("Volume", 1f); } catch { } }

            // set sound Alert 1
            try { sb.SelectedSound = "Alert 1"; } catch { try { sb.SetValue("SelectedSound", "Alert 1"); } catch { } }

            // play
            try { sb.Play(); } catch { try { ((IMyTerminalBlock)sb).ApplyAction("PlaySound"); } catch { } }
        }
        catch { }
    }
}

void StopAllSounds()
{
    var soundBlocks = new List<IMySoundBlock>();
    GridTerminalSystem.GetBlocksOfType(soundBlocks, b =>
        b != null &&
        b.CubeGrid == masterBlock.CubeGrid &&
        b.CustomData != null &&
        b.CustomData.ToUpper().Contains(SERSOUND_CDATA.ToUpper())
    );

    foreach (var sb in soundBlocks)
    {
        try
        {
            // stop
            try { sb.Stop(); } catch { ((IMyTerminalBlock)sb).ApplyAction("StopSound"); }
        }
        catch { }
    }
}

// check if thruster is able to provide thrust (functional, and has max effective thrust > 0)
bool ThrusterIsActive(IMyThrust t)
{
    try
    {
        return t != null && t.IsFunctional && t.IsWorking && t.Enabled && t.MaxEffectiveThrust > 0f;
    }
    catch
    {
        return false;
    }
}

string FormatPlanetResult(PlanetData planet, double upMax, double massKg)
{
    double weight = massKg * planet.Gravity;
    double twr = weight > 0.0 ? upMax / weight : 0.0;

    string status;
    if (twr > 1.2)
        status = "Ok";
    else if (twr >= 1.05)
        status = "Not Recommended";
    else
        status = "Not Ok";

    return planet.Name + ": " + status + " (" + twr.ToString("0.00") + ")";
}
